<script>
document.addEventListener('DOMContentLoaded', function() {
  const contentDisplay = document.getElementById('contentDisplay');
  const loadingScreen = document.getElementById('loadingScreen');
  
  let input = prompt('Please enter a website or search term:');
  
  if (input === null || input.trim() === '') {
    alert('Invalid input or empty input. Please try again.');
  } else {
    let url;
    if (!/\.\S+/.test(input)) {
      url = `https://www.bing.com/search?q=${encodeURIComponent(input.trim())}`;
    } else {
      if (!input.startsWith("http://") && !input.startsWith("https://")) {
        input = 'https://' + input.trim();
      }
      url = input;
    }
    contentDisplay.data = `https://blazecreates.github.io/webpage/index.html?url=${encodeURIComponent(url)}`;
    
    contentDisplay.onload = function() {
      loadingScreen.style.display = 'none';
    };
  }
});
</script>
